## 起泡法排序
```C
#include <stdio.h>
int main()
{
    int a[10];
    int i,j,t;
    //j 0-9循环，再i 9-j循环，只要前一个大于后一个，就换位置
    for (j = 0;j < 9;j++)
    {
        for(i = 0;i < 9 - j;i++)
        {
            if(a[i] > a[i+1])
            {	
                t = a[i];
                a[i] = a[i+1];
                a[i+1] = t;
            }
        }	
    }	
} 
```

void sta(char str[],int *a, int *b, int *c, int *d)
{
    for (int i = 0;str[i] != '\0';i++)
    {
        if  (('A' <= str[i] && str[i] <= 'Z') || ('a' <= str[i] && str[i] <= 'z'))
        (*a)++;
        else if ('0' <= str[i] && str[i] <= '9')
        (*b)++;
        else if (str[i] == ' ')
        (*c)++;
        else
        (*d)++;
    }

## ==静态链表 + 结构体==
```C
struct Student
{
    int num;
    float score;
    struct Student *next;   //静态链表建立（递归）
};

int main()
{
    struct Student a, b, c, *head, *p;
    a.num = 01;
    // 赋值......

    // 连起来链表（递归）
    head = &a;
    a.next = &b;
    b.next = &c;
    c.next = NULL;
    p = head;

    while(p != NULL)
    {
        printf("%d %5.1f\n",p->num, p->score);
        p = p->next;    // p 指向下一结点
    }
    return 0;
}
```
