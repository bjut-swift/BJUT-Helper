```C
    solve_quadratic(a, b, c);
    return 0;
}

void solve_quadratic(double a, double b, double c) {
    if (a == 0) {
        if (b == 0) {
            printf("无解\n");
        } else {
            printf("方程只有一个解: x = %.2f\n", -c / b);
        }
        return;
    }
    
    double delta = b * b - 4 * a * c;
    if (delta > 0) {
        double x1 = (-b + sqrt(delta)) / (2 * a);
        double x2 = (-b - sqrt(delta)) / (2 * a);
        printf("方程有两个实数解: x1 = %.2f, x2 = %.2f\n", x1, x2);
    } else if (delta == 0) {
        double x = -b / (2 * a);
        printf("方程有一个实数解: x = %.2f\n", x);
    } else {
        double realPart = -b / (2 * a);
        double imagPart = sqrt(-delta) / (2 * a);
        printf("方程有两个复数解: x1 = %.2f + %.2fi, x2 = %.2f - %.2fi\n", realPart, imagPart, realPart, imagPart);
    }
}
```

```C
#define N 10

int main()

    int pos = 0;   // 当前报数人位置
    int exit = 3;

    // 初始化：每个人的编号是 1 ~ N

                a[pos] = 0;
                count--;   // 少一个人

            }
        }
        pos = pos + 1;
        if (pos >= N)
        pos = 0;
    }

    // 找出剩下的那个人
    for (i = 0; i < N; i++)
    {
        if (a[i] != 0)
        {
            printf("最后剩下的是原来第 %d 号\n", a[i]);
```

## 排序
### 起泡法排序

- 每次相邻两个数比较，大的往后换；
- 每一轮把剩下最大的数沉到最后；

最多需要 n-1 轮。
#### ==例：对10个数排序==
```C
#include <stdio.h>
int main()
{
	int a[10];
	int i,j,t;
	for (i = 0;i < 10;i++)
		scanf("%d",&a[i]);  //输入排序数
	printf("\n");

    for (j = 0; j < 9; j++)   // 外层循环，控制排序进行9轮（数组有10个元素，只需要9轮），每一轮都把当前未排好元素中最大的元素沉到末尾
    {
        for (i = 0; i < 9 - j; i++)  // 内层循环，每轮比较**相邻**元素；每完成一轮外层循环，内层少比较一个元素
        {
            if (a[i] > a[i+1])        // 如果前一个元素比后一个元素大（顺序不对）
            {   
                t = a[i];             // 用临时变量t保存a[i]的值
                a[i] = a[i+1];        // 将a[i+1]的值赋给a[i]
                a[i+1] = t;           // 把t（原a[i]的值）赋给a[i+1]，完成交换
            }
        }   
    }

	for(i = 0;i < 10;i++)   //输出排序结果
	printf("%d ",a[i]);
	printf("\n");
	return 0;
} 
```

### 选择排序法

先将10个数中最小的数与a[0]对换;再将a[1]~a[9]中最小的数与a[1]对换……每比较一轮,找出一个未经排序的数中最小的一个。共比较9轮。

#### ==例：将10个数从小到大排序==
```C
#include <stdio.h>
int main()
{	void sort(int array[],int n);
    int a[10],i;
    printf("enter array:\n");
    for(i=0;i<10;i++)
        scanf("%d",&a[i]);
    sort(a,10);	//调用sort函数,a为数组名，大小为10
    // 此处传递的是a的指针，sort操作会改变数组a内的值
    printf("The sorted array:\n");
    for(i=0;i<10;i++)
        printf("%d ",a[i]);
    printf("\n");
    return 0;
} 

void sort(int array[],int n)
{	int i,j,k,t;
    for(i=0;i<n-1;i++)  //0 ~ n-1 都确定是由小到大排序了，n也自己确定了
    {	k=i;            //初始值
        for(j=i+1;j<n;j++)
            if(array[j]<array[k])
                k=j;    //i之后所有元素和i比，比i小就换给k
        t=array[k];
        array[k]=array[i];
        array[i]=t;
    }
}

```

```C
void swap_max_last(char str[])
{
    int len = strlen(str);
    if (len <= 1) return; // 长度太短就不处理

    int max_idx = 0; // 假设第一个字符就是最大的

    // 找出ASCII码最大的字符的位置
    for (int i = 1; i < len; i++) {
        if (str[i] > str[max_idx]) {
            max_idx = i;
        }
    }

    // 交换最大字符和最后一个字符

    str[max_idx] = str[len - 1];
    str[len - 1] = temp;
}
```

```C
void process(int *arr, int n)
{
    int *min = arr;
    int *max = arr;

    for (int *p = arr; p < arr + n; p++)
    {
        if (*p < *min)
            min = p;
    }

    int temp = *arr;
    *arr = *min;
    *min = temp;

            max = p;
    }
```

```C
struct Student
{
    int num;
    float score;
    struct Student *next;   //静态链表建立（递归）
};

int main()
{
    struct Student a, b, c, *head, *p;
    a.num = 01;
    // 赋值......

    // 连起来链表（递归）
    head = &a;
    a.next = &b;
    b.next = &c;
    c.next = NULL;
    p = head;

    while(p != NULL)
    {
        printf("%d %5.1f\n",p->num, p->score);
        p = p->next;    // p 指向下一结点
    }
    return 0;
}
```

#### 交换结构体元素顺序
```C
struct Student
{
    int num;
    char name[20];
    float score;
}stu[3] = {
    {01, "aaa", 69},
    {02, "bbb", 78},
    {03, "ccc", 99}
};

int i, j, k;
int n = 5;
struct Student temp;
for(i = 1;i < n - 1;i++)
{
    k = i;
    for(j = i + 1;j < n;j++)
    {
        if(stu[j].score > stu[k].score)
        k = j;
    }
    temp = stu[k];
    stu[k] = stu[i];
    stu[i] = temp;
}
```
